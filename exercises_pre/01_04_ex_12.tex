{For $f(x) = 3-4x$ 

\begin{multicols}{2}
\begin{itemize}
\item $f(3) = -9$
\item $f(-1) = 7$
\item $f\left(\frac{3}{2} \right) = -3$
\item  $f(4x) = 3-16x$
\item $4f(x) = 12-16x$
\item $f(-x) = 4x+3$
\item  $f(x-4) = 19-4x$
\item $f(x) - 4 = -4x-1$
\item  $f\left(x^2\right) = 3-4x^2$
\end{itemize}
\end{multicols}
