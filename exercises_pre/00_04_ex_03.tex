{$z = i$, $w = -1+2i$}
{For  $z = i$ and $w = -1+2i$

\begin{itemize}

\item $z+w = -1+3i$

\item $zw = -2-i$

\item $z^2 = -1$

\item $\frac{1}{z} = -i$

\item $\frac{z}{w} = \frac{2}{5} - \frac{1}{5} \, i$

\item $\frac{w}{z} = 2+i$

\item $\overline{z} = -i$

\item $z\overline{z} = 1$

\item $(\overline{z})^2 = -1$

\end{itemize}
}
