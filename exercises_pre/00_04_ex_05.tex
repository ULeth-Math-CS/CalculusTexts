{$z = 3-5i$, $w = 2+7i$}
{For  $z = 3-5i$ and $w = 2+7i$

\begin{itemize}

\item $z+w = 5+2i$

\item $zw = 41+11i$

\item $z^2 = -16-30i$

\item $\frac{1}{z} = \frac{3}{34} + \frac{5}{34} \,i$

\item $\frac{z}{w} = -\frac{29}{53} - \frac{31}{53} \, i$

\item $\frac{w}{z} = -\frac{29}{34} + \frac{31}{34} \,i$

\item $\overline{z} = 3+5i$

\item $z\overline{z} = 34$

\item $(\overline{z})^2 = -16+30i$

\end{itemize}

}
