{$f(x) = \dfrac{x}{x-1}$}
{For $f(x) = \dfrac{x}{x-1}$ 

\begin{multicols}{2}
\begin{itemize}
\item $f(3) = \dfrac{3}{2}$
\item $f(-1) = \dfrac{1}{2}$
\item $f\left(\dfrac{3}{2} \right) = 3$
\item  $f(4x) = \dfrac{4x}{4x-1}$
\item $4f(x) = \dfrac{4x}{x-1}$
\item $f(-x) = \dfrac{x}{x+1}$
\item  $f(x-4) = \dfrac{x-4}{x-5}$
\item $f(x) - 4 = \dfrac{x}{x-1} - 4 = \frac{4-3x}{x-1}$
\item  $f\left(x^2\right) = \dfrac{x^2}{x^2-1}$
\end{itemize}
\end{multicols}
}
