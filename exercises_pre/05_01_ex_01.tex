{$f(x) = x^2$, $g(x) = 2x+1$}
{For  $f(x) = x^2$ and $g(x) = 2x+1$,
\begin{multicols}{2}

\begin{itemize}

\item  $(g\circ f)(0) = 1$

\item  $(f\circ g)(-1) = 1$

\item  $(f \circ f)(2) = 16$

\item  $(g\circ f)(-3) = 19$

\item  $(f\circ g)\left(\frac{1}{2}\right) = 4$

\item  $(f \circ f)(-2) = 16$

\end{itemize}

\end{multicols}}
