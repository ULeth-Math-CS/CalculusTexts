{$f(x) = \dfrac{2}{x}$}
{For $f(x) = \frac{2}{x}$

\begin{itemize}
\item  $f(2) = 1$
\item  $f(-2) = -1$
\item  $f(2a) = \frac{1}{a}$
\item  $2 f(a) = \frac{4}{a}$
\item $f(a+2) = \frac{2}{a+2}$
\item $f(a) + f(2) = \frac{2}{a}+1 = \frac{a+2}{2}$
\item  $f \left( \frac{2}{a} \right) = a$
\item $\frac{f(a)}{2} =  \frac{1}{a}$
\item  $f(a + h) = \frac{2}{a+h}$
\end{itemize}
}
