{Let ${\displaystyle f(x) = \left\{ \begin{array}{rcr}
x^{2} & \mbox{ if } & x \leq -1\\
\sqrt{1 - x^{2}} & \mbox{ if } & -1 < x \leq 1\\
x & \mbox{ if } & x > 1  \end{array} \right. }$  Compute the following function values.
\begin{multicols}{2}
\begin{enumerate}
\item $f(4)$
\item $f(-3)$
\item $f(1)$
\item $f(0)$
\item $f(-1)$
\item $f(-0.999)$
\end{enumerate}
\end{multicols}
}
{\begin{enumerate}
\item $f(4) = 4$
\item $f(-3) = 9$
\item $f(1) = 0$
\item $f(0) = 1$
\item $f(-1) = 1$
\item $f(-0.999) \approx 0.0447$
\end{enumerate}}
