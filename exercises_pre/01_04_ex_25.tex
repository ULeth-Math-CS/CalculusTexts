{$f(x) = \dfrac{x}{2}$}
{For $f(x) = \frac{x}{2}$

\begin{itemize}
\item  $f(2) = 1$
\item  $f(-2) = -1$
\item  $f(2a) = a$
\item  $2 f(a) = a$
\item $f(a+2) = \frac{a+2}{2}$
\item $f(a) + f(2) = \frac{a}{2}+ 1 = \frac{a+2}{2}$
\item  $f \left( \frac{2}{a} \right) = \frac{1}{a}$
\item $\frac{f(a)}{2} =  \frac{a}{4}$
\item  $f(a + h) = \frac{a+h}{2}$
\end{itemize}
}
