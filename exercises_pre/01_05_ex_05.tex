{$f(x) = \sqrt{x+3}$ and  $g(x) = 2x-1$}
{For $f(x) = \sqrt{x+3}$ and  $g(x) = 2x-1$

\begin{multicols}{2}
\begin{itemize}
\item  $(f+g)(2) = 3+\sqrt{5}$
\item  $(f-g)(-1) = 3+\sqrt{2}$
\item  $(g-f)(1) = -1$
\item  $(fg)\left(\frac{1}{2}\right) = 0$
\item  $\left(\frac{f}{g}\right)(0) = -\sqrt{3}$
\item  $\left(\frac{g}{f}\right)\left(-2\right) = -5$
\end{itemize}
\end{multicols}
}
