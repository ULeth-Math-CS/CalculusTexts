\item  $(g \circ f)(x) = \sqrt{x^2-x-6}$, domain: $(-\infty, -2] \cup [3,\infty)$

\item  $(f \circ g)(x) =x-6-\sqrt{x-5}$, domain: $[5,\infty)$

\item  $(f \circ f)(x) =x^4-2x^3-2x^2+3x+1$, domain: $(-\infty, \infty)$

\end{itemize}}
