{$z = 2+3i$, $w = 4i$}
{For $z = 2+3i$ and $w = 4i$

\begin{itemize}

\item $z+w = 2+7i$

\item $zw = -12+8i$

\item $z^2 = -5 + 12i$

\item $\frac{1}{z} = \frac{2}{13} - \frac{3}{13} \, i$

\item $\frac{z}{w} = \frac{3}{4} - \frac{1}{2} \, i$

\item $\frac{w}{z} = \frac{12}{13} + \frac{8}{13} \,i$

\item $\overline{z} = 2-3i$

\item $z\overline{z} = 13$

\item $(\overline{z})^2 = -5-12i$

\end{itemize}

}
