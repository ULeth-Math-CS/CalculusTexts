{$z = -\dfrac{\sqrt{2}}{2} + \dfrac{\sqrt{2}}{2} \, i$, $w = -\dfrac{\sqrt{2}}{2} - \dfrac{\sqrt{2}}{2} \, i$}
{For   $z = -\frac{\sqrt{2}}{2} + \frac{\sqrt{2}}{2} \, i$ and $w = -\frac{\sqrt{2}}{2} - \frac{\sqrt{2}}{2} \, i$

\begin{itemize}

\item $z + w = -\sqrt{2}$

\item $zw = 1$

\item $z^2 =-i$

\item $\frac{1}{z} = -\frac{\sqrt{2}}{2} - \frac{\sqrt{2}}{2} \, i$

\item $\frac{z}{w} = -i$

\item $\frac{w}{z} = i$

\item $\overline{z} = -\frac{\sqrt{2}}{2} - \frac{\sqrt{2}}{2} \, i$

\item $z\overline{z} = 1$

\item $(\overline{z})^2 = i$

\end{itemize}
}
