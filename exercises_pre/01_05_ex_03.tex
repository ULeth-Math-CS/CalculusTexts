{For  $f(x) = x^2 - x$ and  $g(x) = 12-x^2$

\begin{multicols}{2}
\begin{itemize}
\item  $(f+g)(2) = 10$
\item  $(f-g)(-1) = -9$
\item  $(g-f)(1) = 11$
\item  $(fg)\left(\frac{1}{2}\right) = -\frac{47}{16}$
\item  $\left(\frac{f}{g}\right)(0) = 0$
\item  $\left(\frac{g}{f}\right)\left(-2\right) = \frac{4}{3}$
\end{itemize}
\end{multicols}
