{$f(x) = 5-2x$}
{\item For $f(x) = 5-2x$


\begin{itemize}
\item  $f(2) = 1$
\item  $f(-2) = 9$
\item  $f(2a) = 5-4a$
\item  $2 f(a) = 10-4a$
\item $f(a+2) = 1-2a$
\item $f(a) + f(2) = 6-2a$
\item  $f \left( \frac{2}{a} \right) = 5 - \frac{4}{a} = \frac{5a-4}{a}$
\item $\frac{f(a)}{2} = \frac{5-2a}{2}$
\item  $f(a + h) = 5-2a-2h$
\end{itemize}
}
