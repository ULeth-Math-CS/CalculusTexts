{$f(x) = |x-1|$, $g(x) = x^2-5$}
{ For   $f(x) = |x-1|$ and $g(x) = x^2-5$,
\begin{multicols}{2}

\begin{itemize}

\item  $(g\circ f)(0) = -4$

\item  $(f\circ g)(-1) = 5$

\item  $(f \circ f)(2) = 0$

\item  $(g\circ f)(-3) = 11$

\item  $(f\circ g)\left(\frac{1}{2}\right) = \frac{23}{4}$

\item  $(f \circ f)(-2) = 2$

\end{itemize}

\end{multicols}}
