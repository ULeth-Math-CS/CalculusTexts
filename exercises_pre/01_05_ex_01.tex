{$f(x) = 3x+1$ and  $g(x) = 4-x$}
{For  $f(x) = 3x+1$ and $g(x) = 4-x$

\begin{multicols}{2}
\begin{itemize}

\item  $(f+g)(2) = 9$
\item  $(f-g)(-1) = -7$
\item  $(g-f)(1) = -1$
\item  $(fg)\left(\frac{1}{2}\right) = \frac{35}{4}$
\item  $\left(\frac{f}{g}\right)(0) = \frac{1}{4}$
\item  $\left(\frac{g}{f}\right)\left(-2\right) = -\frac{6}{5}$

\end{itemize}
\end{multicols}
}
