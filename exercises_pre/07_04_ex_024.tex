{If $\sin(\alpha) = \dfrac{3}{5}$, where $0 < \alpha < \dfrac{\pi}{2}$, and $\cos(\beta) = \dfrac{12}{13}$ where $\dfrac{3\pi}{2} < \beta < 2\pi$, find 

\begin{enumerate}
\item $\sin(\alpha + \beta)$
\item $\cos(\alpha - \beta)$
\item $\tan(\alpha - \beta)$
\end{enumerate}
}
{
\begin{enumerate}
\item $\sin(\alpha + \beta) = \dfrac{16}{65}$
\item $\cos(\alpha - \beta) = \dfrac{33}{65}$
\item $\tan(\alpha - \beta) = \dfrac{56}{33}$
\end{enumerate}
}
