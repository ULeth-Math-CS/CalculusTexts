{For $f(x) = 2x+1$ 

\begin{multicols}{2}
\begin{itemize}
\item $f(3) = 7$
\item $f(-1) = -1$
\item $f\left(\frac{3}{2} \right) = 4$
\item  $f(4x) = 8x+1$
\item $4f(x) = 8x+4$
\item $f(-x) = -2x+1$
\item  $f(x-4) = 2x-7$
\item $f(x) - 4 = 2x-3$
\item  $f\left(x^2\right) = 2x^2+1$
\end{itemize}
\end{multicols}}
