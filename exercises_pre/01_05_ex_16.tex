{For $f(x) = -x^2+x+6$ and $g(x) = x^2-9$

\begin{itemize}
\item $(f+g)(x) = x-3$
      Domain: $(-\infty, \infty)$
\item $(f-g)(x) = -2x^2+x+15$
      Domain:  $(-\infty, \infty)$
\item $(fg)(x) = -x^4+x^3+15x^2-9x-54$
      Domain: $(-\infty, \infty)$
\item $\left(\frac{f}{g}\right)(x) = -\frac{x+2}{x+3}$
      Domain:  $\left(-\infty, -3 \right) \cup \left(-3, 3 \right) \cup (3, \infty)$
\end{itemize}
