{$z = 1+i$, $w = -i$}
{For $z = 1+i$ and $w = -i$

\begin{itemize}

\item $z+w = 1$

\item $zw = 1-i$

\item $z^2 = 2i$

\item $\frac{1}{z} = \frac{1}{2} - \frac{1}{2} \, i$

\item $\frac{z}{w} = -1+i$

\item $\frac{w}{z} = -\frac{1}{2} - \frac{1}{2} \, i$

\item $\overline{z} = 1-i$

\item $z\overline{z} = 2$

\item $(\overline{z})^2 = -2i$

\end{itemize}
}
