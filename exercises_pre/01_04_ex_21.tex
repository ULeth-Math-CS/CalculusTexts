{For $f(x) = 2x^2-1$

\begin{itemize}
\item  $f(2) = 7$
\item  $f(-2) = 7$
\item  $f(2a) = 8a^2-1$
\item  $2 f(a) = 4a^2-2$
\item $f(a+2) = 2a^2+8a+7$
\item $f(a) + f(2) = 2a^2+6$
\item  $f \left( \frac{2}{a} \right) = \frac{8}{a^2} - 1 = \frac{8-a^2}{a^2}$
\item $\frac{f(a)}{2} =  \frac{2a^2-1}{2}$
\item  $f(a + h) = 2a^2+4ah+2h^2-1$
\end{itemize}
}
