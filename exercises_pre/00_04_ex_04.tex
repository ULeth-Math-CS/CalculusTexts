{$z = 4i$, $w = 2-2i$}
{ For  $z = 4i$ and $w = 2-2i$

\begin{itemize}

\item $z+w = 2+2i$

\item $zw = 8+8i$

\item $z^2 = -16$

\item $\frac{1}{z} = -\frac{1}{4} \,i$

\item $\frac{z}{w} = -1+i$

\item $\frac{w}{z} = -\frac{1}{2} - \frac{1}{2} \,i$

\item $\overline{z} = -4i$

\item $z\overline{z} = 16$

\item $(\overline{z})^2 = -16$

\end{itemize}

}
